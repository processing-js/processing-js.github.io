Name

alpha()
Examples 
noStroke();

color c = color(0, 126, 255, 102);

fill(c);

rect(15, 15, 35, 70);

float value = alpha(c);  //Sets "value" to "102"

fill(value);

rect(50, 15, 35, 70);

Syntax
