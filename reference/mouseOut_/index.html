Name

mouseOut()
Examples 
// Moving the mouse into and out of the canvas

// changes the background color



void setup() {

  size(200,200);

  background(10);

}



void draw() {}



void mouseOver() {

  background(50,80,140);

}



void mouseOut() {

  background(140,50,80);

}
Description The mouseOut() function is called when the mouse pointer leaves a sketch.
